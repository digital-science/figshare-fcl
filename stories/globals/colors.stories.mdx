export const onSaveTitleClipboard = (ev) => {
  const dummy = document.createElement("textarea");
  const { title } = ev.target;
  document.body.appendChild(dummy);
  dummy.value = title;
  dummy.select();
  document.execCommand("copy");
  document.body.removeChild(dummy);
};

## Overview

Color definitions and style guides. The colors we use are exported as `css variables` and can be imported directly (`styles/settings/colors.css`) and used or as part of the global stylesheet (`styles/index.css`)

You can import the color variables in your code:
```jsx
// import only color variables
import "@digital-science/figshare-fcl/styles/settings/colors.css";
// or with typography and other styles as well
import "@digital-science/figshare-fcl/styles/index.css";
```
And reference the variables in your css files:
```css
.text {
  color: var(--color-typography-primary);
}

.secondaryText {
  color: var(--color-typography-secondary);
}

.error-text {
  color: var(--color-typography-error);
}
```
